# 递归

[70. Climbing Stairs](https://leetcode.com/problems/climbing-stairs/)

```
package main

import (
	"fmt"
)

func main() {
	num := 40
	result := func2(num)
	fmt.Println(result)
}

func func1(num int) int {
	if num == 1 {
		return 1
	}

	if num == 2 {
		return 2
	}

	return func1(num-1) + func1(num-2)
}

func func2(num int) int {
	fmt.Println("func2")
	if num == 1 {
		return 1
	}

	if num == 2 {
		return 2
	}

	// when num >= 3
	a := 1
	b := 2
	var temp int

	for i:=3; i <= num; i++ {
			temp = a + b
			a = b
			b = temp
	}
	return temp
}
```

```
// Bubble Sort
func BubbleSort(list []int) []int {
	if len(list) == 0 || len(list) == 1 {
		return list
	}

	for i := 0; i < len(list); i++ {
		var flag bool
		for j := 0; j < len(list)-i-1; j++ {
			if list[j] > list[j+1] {
				temp := list[j]
				list[j] = list[j+1]
				list[j+1] = temp
				flag = true
			}
		}
		if flag == false {
			break
		}
	}

	return list
}

// Insert Sort
func InsertSort(list []int) []int {
	if len(list) == 0 || len(list) == 1 {
		return list
	}

	for i:= 1;i< len(list);i++ {
		value := list[i]
		j := i -1
		for ;j >= 0; j-- {
			if list[j] > value {
				list[j+1] = list[j]
			} else {
				break
			}
		}
		list[j+1] = value
	}

	return list
}
```

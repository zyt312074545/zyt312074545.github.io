[70. Climbing Stairs](https://leetcode.com/problems/climbing-stairs/)

```
package main

import (
	"fmt"
)

func main() {
	num := 40
	result := func2(num)
	fmt.Println(result)
}

func func1(num int) int {
	if num == 1 {
		return 1
	}

	if num == 2 {
		return 2
	}

	return func1(num-1) + func1(num-2)
}

func func2(num int) int {
	fmt.Println("func2")
	if num == 1 {
		return 1
	}

	if num == 2 {
		return 2
	}

	// when num >= 3
	a := 1
	b := 2
	var temp int

	for i:=3; i <= num; i++ {
			temp = a + b
			a = b
			b = temp
	}
	return temp
}
```

```
// Bubble Sort
func BubbleSort(list []int) []int {
	if len(list) == 0 || len(list) == 1 {
		return list
	}

	for i := 0; i < len(list); i++ {
		var flag bool
		for j := 0; j < len(list)-i-1; j++ {
			if list[j] > list[j+1] {
				temp := list[j]
				list[j] = list[j+1]
				list[j+1] = temp
				flag = true
			}
		}
		if flag == false {
			break
		}
	}

	return list
}

// Insert Sort
func InsertSort(list []int) []int {
	if len(list) == 0 || len(list) == 1 {
		return list
	}

	for i:= 1;i< len(list);i++ {
		value := list[i]
		j := i -1
		for ;j >= 0; j-- {
			if list[j] > value {
				list[j+1] = list[j]
			} else {
				break
			}
		}
		list[j+1] = value
	}

	return list
}

// Select Sort
func SelectSort(list []int) []int {
	if len(list) < 2 {
		return list
	}

	fmt.Println("Begin Select Sort")
	for i := 0; i < len(list)-1; i++ {
		minIndex := i
		minValue := list[i]
		for j := i + 1; j < len(list); j++ {
			if minValue > list[j] {
				minIndex = j
				minValue = list[j]
			}
		}

		if minIndex != i {
			list[minIndex] = list[i]
			list[i] = minValue
		}
	}
	fmt.Println("End Select Sort")

	return list
}

// Merge Sort
func MergeSort(list []int) []int {
	if len(list) < 2 {
		return list
	}

	p := int(len(list) / 2)
	leftResult := MergeSort(list[:p])
	rightResult := MergeSort(list[p:])
	result := merge(leftResult, rightResult, len(list))

	return result
}

func merge(leftList, rightList []int, length int) []int {
	temp := make([]int, 0, length)
	
	for len(leftList) > 0 || len(rightList) > 0 {
		if len(leftList) == 0 {
			return append(temp, rightList...)
		}
		if len(rightList) == 0 {
			return append(temp, leftList...)
		}
		if leftList[0] <= rightList[0] {
			temp = append(temp, leftList[0])
			leftList = leftList[1:]
		} else {
			temp = append(temp, rightList[0])
			rightList = rightList[1:]
		}
	}

	return temp
}

// Quick Sort
func QuickSort(list []int) []int {
	if len(list) <= 1 {
		return list
	}

	// 设置左右锚点
	left, right := 0, len(list)-1
	for i := range list {
		if list[i] < list[right] {
			list[i], list[left] = list[left], list[i]
			left++
		}
	}
	list[left], list[right] = list[right], list[left]

	QuickSort(list[:left])
	QuickSort(list[left+1:])

	return list
}
```
